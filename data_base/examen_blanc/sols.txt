select * from (select mainom,maiid,count(*) as cf from maisons inner join personnages on permaiid=maiid where persexe ="f" group by maiid,persexe) as t2 inner join (select maiid as t1id, count(*) as ch from maisons inner join personnages on permaiid=maiid where persexe ="h" group by maiid,persexe) as t1 on t1id=maiid where cf>ch;
